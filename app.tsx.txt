import React, { useState, useEffect, useRef } from 'react';
import { fetchMarketSnapshot } from './services/geminiService';
import { MarketSnapshotData } from './types';
import MarketSnapshotReport from './components/MarketSnapshotReport';
import * as htmlToImage from 'html-to-image';

const App: React.FC = () => {
  const [data, setData] = useState<MarketSnapshotData | null>(null);
  const [loading, setLoading] = useState<boolean>(true);
  const [error, setError] = useState<string | null>(null);
  const [exporting, setExporting] = useState<boolean>(false);
  const [showToast, setShowToast] = useState<boolean>(false);
  const reportRef = useRef<HTMLDivElement>(null);

  const loadData = async () => {
    setLoading(true);
    setError(null);
    try {
      const result = await fetchMarketSnapshot();
      setData(result);
    } catch (err: any) {
      console.error("Failed to fetch market data:", err);
      setError(err?.message || "Connection to Market Intel terminal failed. Please check your API key.");
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    loadData();
  }, []);

  const copyImageToClipboard = async () => {
    if (reportRef.current === null) return;
    setExporting(true);
    try {
      // 3x pixel ratio ensures text is sharp on high-DPI smartphone screens
      // This is crucial for WhatsApp which compresses images
      const dataUrl = await htmlToImage.toPng(reportRef.current, { 
        quality: 1.0, 
        pixelRatio: 3,
        backgroundColor: '#ffffff',
        cacheBust: true,
      });
      
      const res = await fetch(dataUrl);
      const blob = await res.blob();
      const item = new ClipboardItem({ 'image/png': blob });
      await navigator.clipboard.write([item]);
      
      setShowToast(true);
      setTimeout(() => setShowToast(false), 4000);
    } catch (err) {
      console.error('Clipboard copy failed:', err);
      alert('Unable to copy automatically. Please ensure you are using a secure (HTTPS) connection or try clicking "Sync Data" and trying again.');
    } finally {
      setExporting(false);
    }
  };

  if (loading) {
    return (
      <div className="flex flex-col items-center justify-center min-h-screen bg-slate-50">
        <div className="relative w-24 h-24 mb-8">
           <div className="absolute inset-0 border-4 border-slate-200 rounded-full"></div>
           <div className="absolute inset-0 border-4 border-orange-500 rounded-full border-t-transparent animate-spin"></div>
        </div>
        <div className="text-center">
          <p className="text-2xl font-black text-slate-900 tracking-tighter uppercase mb-1">Market Intel Terminal</p>
          <p className="text-[10px] text-slate-400 font-bold uppercase tracking-[0.3em]">Vivek Financial Focus Ltd.</p>
        </div>
      </div>
    );
  }

  if (error) {
    return (
      <div className="flex flex-col items-center justify-center min-h-screen bg-slate-50 p-6 text-center">
        <div className="bg-white p-12 rounded-[2.5rem] shadow-2xl max-w-md border border-slate-200">
          <div className="w-16 h-16 bg-rose-50 text-rose-500 rounded-full flex items-center justify-center mx-auto mb-6">
            <i className="fa-solid fa-triangle-exclamation text-2xl"></i>
          </div>
          <h2 className="text-2xl font-black text-slate-900 mb-3 uppercase tracking-tighter">System Offline</h2>
          <p className="text-slate-500 mb-8 font-medium leading-relaxed">{error}</p>
          <button 
              onClick={loadData}
              className="w-full bg-[#0f172a] hover:bg-slate-800 text-white font-black py-4 px-8 rounded-2xl transition-all shadow-xl uppercase tracking-widest text-[10px]"
          >
              Attempt Reconnect
          </button>
        </div>
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-slate-100 pb-20 font-sans">
      {showToast && (
        <div className="fixed top-8 left-1/2 -translate-x-1/2 z-50 bg-[#0f172a] text-white px-8 py-4 rounded-full shadow-2xl flex items-center gap-3 animate-bounce">
          <i className="fa-solid fa-circle-check text-emerald-400"></i>
          <div className="font-black uppercase tracking-widest text-[10px]">Copied! Open WhatsApp & Paste</div>
        </div>
      )}

      {exporting && (
        <div className="fixed inset-0 z-50 bg-[#0f172a]/90 backdrop-blur-md flex flex-col items-center justify-center text-white text-center p-6">
            <i className="fa-solid fa-camera-retro text-5xl text-orange-500 mb-6 animate-pulse"></i>
            <p className="text-xl font-black uppercase tracking-tighter">Rendering Institutional Snapshot...</p>
            <p className="text-[10px] font-bold text-slate-400 uppercase tracking-widest mt-2">Optimizing for WhatsApp HD Quality</p>
        </div>
      )}

      <nav className="sticky top-0 z-40 bg-white/80 backdrop-blur-xl border-b border-slate-200 px-6 py-4">
        <div className="max-w-6xl mx-auto flex flex-col md:flex-row items-center justify-between gap-4">
          <div className="flex items-center gap-3">
             <div className="bg-[#0f172a] text-white w-10 h-10 flex items-center justify-center rounded-xl shadow-lg">
               <i className="fa-solid fa-bolt text-orange-400"></i>
             </div>
             <div>
               <h1 className="text-xl font-black text-slate-900 leading-none uppercase tracking-tighter">SnapReport Pro</h1>
               <p className="text-[8px] text-slate-400 font-black uppercase tracking-[0.2em] mt-1">Vivek Financial Intelligence</p>
             </div>
          </div>
          
          <div className="flex items-center gap-3">
            <button 
              onClick={loadData}
              className="bg-white hover:bg-slate-50 text-slate-900 font-black py-2.5 px-5 rounded-xl transition-all border border-slate-200 uppercase tracking-widest text-[10px]"
            >
              <i className="fa-solid fa-rotate mr-2"></i>
              Sync Data
            </button>
            <button 
              onClick={copyImageToClipboard}
              className="bg-emerald-600 hover:bg-emerald-700 text-white font-black py-2.5 px-6 rounded-xl shadow-lg shadow-emerald-600/20 transition-all uppercase tracking-widest text-[10px] flex items-center gap-2"
            >
              <i className="fa-brands fa-whatsapp text-lg"></i>
              Copy for WhatsApp
            </button>
          </div>
        </div>
      </nav>

      <main className="max-w-6xl mx-auto mt-12 flex flex-col items-center px-6">
        <div className="p-4 bg-white rounded-[3rem] shadow-[0_32px_80px_-20px_rgba(0,0,0,0.1)] border border-slate-100">
          {data && <MarketSnapshotReport data={data} reportRef={reportRef} />}
        </div>
        
        <div className="mt-12 mb-20 flex items-center gap-8 opacity-20 grayscale pointer-events-none">
            <div className="flex items-center gap-3">
                <div className="w-8 h-8 border-2 border-slate-900 rounded-full flex items-center justify-center text-[10px] font-black">1</div>
                <span className="text-[10px] font-black uppercase">Copy</span>
            </div>
            <i className="fa-solid fa-chevron-right text-[10px]"></i>
            <div className="flex items-center gap-3">
                <div className="w-8 h-8 border-2 border-slate-900 rounded-full flex items-center justify-center text-[10px] font-black">2</div>
                <span className="text-[10px] font-black uppercase">WhatsApp</span>
            </div>
            <i className="fa-solid fa-chevron-right text-[10px]"></i>
            <div className="flex items-center gap-3">
                <div className="w-8 h-8 border-2 border-slate-900 rounded-full flex items-center justify-center text-[10px] font-black">3</div>
                <span className="text-[10px] font-black uppercase">Paste</span>
            </div>
        </div>
      </main>
    </div>
  );
};

export default App;
