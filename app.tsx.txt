import React, { useState, useEffect, useRef } from 'react';
import { fetchMarketSnapshot } from './services/geminiService';
import { MarketSnapshotData } from './types';
import MarketSnapshotReport from './components/MarketSnapshotReport';
import * as htmlToImage from 'html-to-image';

const App: React.FC = () => {
  const [data, setData] = useState<MarketSnapshotData | null>(null);
  const [loading, setLoading] = useState<boolean>(true);
  const [error, setError] = useState<string | null>(null);
  const [exporting, setExporting] = useState<boolean>(false);
  const [showToast, setShowToast] = useState<boolean>(false);
  const reportRef = useRef<HTMLDivElement>(null);

  const loadData = async () => {
    setLoading(true);
    setError(null);
    try {
      const result = await fetchMarketSnapshot();
      setData(result);
    } catch (err: any) {
      console.error("Failed to fetch market data:", err);
      setError(err?.message || "Failed to fetch market intelligence. Please check your API key.");
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    loadData();
  }, []);

  const copyImageToClipboard = async () => {
    if (reportRef.current === null) return;
    setExporting(true);
    try {
      // Use higher pixel ratio for mobile clarity
      const dataUrl = await htmlToImage.toPng(reportRef.current, { 
        quality: 1, 
        pixelRatio: 3,
        backgroundColor: '#ffffff'
      });
      const res = await fetch(dataUrl);
      const blob = await res.blob();
      const item = new ClipboardItem({ 'image/png': blob });
      await navigator.clipboard.write([item]);
      setShowToast(true);
      setTimeout(() => setShowToast(false), 3000);
    } catch (err) {
      console.error('Clipboard copy failed:', err);
      alert('Unable to copy directly. Try downloading the image or ensure you are on a secure (HTTPS) connection.');
    } finally {
      setExporting(false);
    }
  };

  if (loading) {
    return (
      <div className="flex flex-col items-center justify-center min-h-screen bg-[#f8fafc]">
        <div className="relative w-32 h-32 mb-10">
           <div className="absolute inset-0 border-[6px] border-slate-200 rounded-full"></div>
           <div className="absolute inset-0 border-[6px] border-orange-500 rounded-full border-t-transparent animate-spin"></div>
        </div>
        <div className="text-center">
          <p className="text-3xl font-black text-slate-900 tracking-tighter uppercase mb-2">Automating Intelligence</p>
          <p className="text-[10px] text-slate-400 font-black uppercase tracking-[0.4em]">Vivek Financial Terminal</p>
        </div>
      </div>
    );
  }

  if (error) {
    return (
      <div className="flex flex-col items-center justify-center min-h-screen bg-[#f8fafc] p-6 text-center">
        <div className="bg-white p-14 rounded-[40px] shadow-2xl max-w-lg border border-slate-200">
          <div className="w-20 h-20 bg-rose-50 text-rose-500 rounded-full flex items-center justify-center mx-auto mb-8">
            <i className="fa-solid fa-triangle-exclamation text-3xl"></i>
          </div>
          <h2 className="text-3xl font-black text-slate-900 mb-4 uppercase tracking-tighter">Connection Error</h2>
          <p className="text-slate-500 mb-10 font-medium leading-relaxed">{error}</p>
          <button 
              onClick={loadData}
              className="w-full bg-[#0f172a] hover:bg-slate-800 text-white font-black py-5 px-10 rounded-2xl transition-all shadow-xl uppercase tracking-widest text-xs"
          >
              Retry Intelligence Sync
          </button>
        </div>
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-slate-50 pb-20 overflow-x-hidden">
      {/* Dynamic Toast */}
      {showToast && (
        <div className="fixed top-10 left-1/2 -translate-x-1/2 z-50 bg-[#0f172a] text-white px-10 py-5 rounded-full shadow-2xl flex items-center gap-4 animate-bounce">
          <i className="fa-solid fa-circle-check text-emerald-400 text-xl"></i>
          <div className="font-black uppercase tracking-widest text-xs">Ready to Paste in WhatsApp</div>
        </div>
      )}

      {/* Processing Overlay */}
      {exporting && (
        <div className="fixed inset-0 z-50 bg-[#0f172a]/80 backdrop-blur-md flex flex-col items-center justify-center text-white">
            <i className="fa-solid fa-camera-retro text-6xl text-orange-500 mb-6 animate-pulse"></i>
            <p className="text-2xl font-black uppercase tracking-tighter">Generating High-Res Export...</p>
        </div>
      )}

      {/* Control Navigation */}
      <div className="sticky top-0 z-40 bg-white/80 backdrop-blur-xl border-b border-slate-200 p-5">
        <div className="max-w-6xl mx-auto flex flex-col md:flex-row items-center justify-between gap-6">
          <div className="flex items-center">
             <div className="bg-[#0f172a] text-white w-12 h-12 flex items-center justify-center rounded-2xl mr-4 shadow-xl">
               <i className="fa-solid fa-bolt-lightning text-xl text-orange-400"></i>
             </div>
             <div>
               <h1 className="text-2xl font-black text-slate-900 leading-none uppercase tracking-tighter">SnapReport Pro</h1>
               <p className="text-[10px] text-slate-400 font-black uppercase tracking-[0.2em] mt-1.5">Vivek Financial Focus Ltd.</p>
             </div>
          </div>
          
          <div className="flex items-center gap-4">
            <button 
              onClick={loadData}
              className="flex items-center gap-2 bg-white hover:bg-slate-50 text-slate-900 font-black py-3.5 px-6 rounded-2xl transition-all border border-slate-200 uppercase tracking-widest text-[10px]"
            >
              <i className="fa-solid fa-rotate"></i>
              Refresh Data
            </button>
            <button 
              onClick={copyImageToClipboard}
              className="flex items-center gap-3 bg-emerald-600 hover:bg-emerald-700 text-white font-black py-3.5 px-8 rounded-2xl shadow-xl shadow-emerald-600/30 transition-all uppercase tracking-widest text-[10px]"
            >
              <i className="fa-brands fa-whatsapp text-lg"></i>
              Copy for WhatsApp
            </button>
          </div>
        </div>
      </div>

      {/* Centered Report View */}
      <div className="max-w-6xl mx-auto mt-16 flex justify-center px-6">
        <div className="p-6 bg-white rounded-[60px] shadow-[0_40px_100px_-20px_rgba(0,0,0,0.15)] border border-slate-100">
          {data && <MarketSnapshotReport data={data} reportRef={reportRef} />}
        </div>
      </div>

      {/* Simplified Helper */}
      <div className="max-w-md mx-auto mt-20 text-center">
        <div className="inline-flex items-center gap-8 text-slate-300">
            <div className="flex flex-col items-center">
                <div className="w-10 h-10 border-2 border-slate-200 rounded-full flex items-center justify-center mb-3 font-black text-xs">1</div>
                <span className="text-[10px] font-black uppercase tracking-widest">Generate</span>
            </div>
            <div className="h-px w-8 bg-slate-200"></div>
            <div className="flex flex-col items-center">
                <div className="w-10 h-10 border-2 border-slate-200 rounded-full flex items-center justify-center mb-3 font-black text-xs">2</div>
                <span className="text-[10px] font-black uppercase tracking-widest">Copy</span>
            </div>
            <div className="h-px w-8 bg-slate-200"></div>
            <div className="flex flex-col items-center">
                <div className="w-10 h-10 bg-emerald-100 border-2 border-emerald-500 text-emerald-600 rounded-full flex items-center justify-center mb-3 font-black text-xs">3</div>
                <span className="text-[10px] font-black uppercase tracking-widest text-emerald-600">WhatsApp</span>
            </div>
        </div>
      </div>
    </div>
  );
};

export default App;
